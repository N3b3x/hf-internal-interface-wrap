629 virtual hf_u32_t SendMessageBatch(const hf_can_message_t* messages, hf_u32_t count,
630 hf_u32_t timeout_ms = 1000) noexcept {
631 if (!messages || count == 0) {
632 return 0;
633 }
634
635 hf_u32_t sent_count = 0;
636 for (hf_u32_t i = 0; i < count; ++i) {
637 if (SendMessage(messages[i], timeout_ms) == hf_can_err_t::CAN_SUCCESS) {
638 sent_count++;
639 }
640 }
641 return sent_count;
642 }
643
652 virtual hf_u32_t ReceiveMessageBatch(hf_can_message_t* messages, hf_u32_t max_count,
653 hf_u32_t timeout_ms = 100) noexcept {
654 if (!messages || max_count == 0) {
655 return 0;
656 }
657
658 hf_u32_t received_count = 0;
659 for (hf_u32_t i = 0; i < max_count; ++i) {
660 if (ReceiveMessage(messages[i], timeout_ms) == hf_can_err_t::CAN_SUCCESS) {
661 received_count++;
662 } else {
663 break; // No more messages available
664 }
665 }
666 return received_count;
667 }
