278#define HF_GPIO_IS_VALID_GPIO(gpio_num) \
279 ((gpio_num) >= 0 && (gpio_num) <= HF_MCU_GPIO_MAX_PIN_NUMBER)
280
281#define HF_GPIO_IS_VALID_PIN(gpio_num) HF_GPIO_IS_VALID_GPIO(gpio_num)
282
283#define HF_GPIO_IS_VALID_OUTPUT_GPIO(gpio_num) \
284 (HF_GPIO_IS_VALID_GPIO(gpio_num) && !HF_GPIO_IS_INPUT_ONLY_PIN(gpio_num))
285
286#define HF_GPIO_IS_VALID_RTC_GPIO(gpio_num) ((gpio_num) >= 0 && (gpio_num) <= 7)
287
288#define HF_GPIO_IS_VALID_LP_IO(gpio_num) ((gpio_num) >= 0 && (gpio_num) <= 7)
289
290#define HF_GPIO_SUPPORTS_ADC(gpio_num) ((gpio_num) >= 0 && (gpio_num) <= 6)
291
292#define HF_GPIO_IS_SPI_FLASH_PIN(gpio_num) ((gpio_num) >= 24 && (gpio_num) <= 30)
293
294#define HF_GPIO_IS_INPUT_ONLY_PIN(gpio_num) (false) /* ESP32 has no input-only pins */
295
296#define HF_GPIO_SUPPORTS_PULL_UP(gpio_num) (HF_GPIO_IS_VALID_GPIO(gpio_num))
297
298#define HF_GPIO_SUPPORTS_PULL_DOWN(gpio_num) (HF_GPIO_IS_VALID_GPIO(gpio_num))
299
300#define HF_GPIO_SUPPORTS_OPEN_DRAIN(gpio_num) (HF_GPIO_IS_VALID_OUTPUT_GPIO(gpio_num))
301
305#define HF_GPIO_TO_ADC_UNIT(gpio_num) (HF_GPIO_SUPPORTS_ADC(gpio_num) ? 1 : 0xFF)
306
307#define HF_GPIO_TO_ADC_CHANNEL(gpio_num) (HF_GPIO_SUPPORTS_ADC(gpio_num) ? (gpio_num) : 0xFF)
308
312#define HF_GPIO_TO_RTC_GPIO(gpio_num) (HF_GPIO_IS_VALID_RTC_GPIO(gpio_num) ? (gpio_num) : 0xFF)
313
314#define HF_GPIO_TO_LP_IO(gpio_num) (HF_GPIO_IS_VALID_LP_IO(gpio_num) ? (gpio_num) : 0xFF)

319#define HF_GPIO_IS_SAFE_FOR_GENERAL_USE(gpio_num) \
