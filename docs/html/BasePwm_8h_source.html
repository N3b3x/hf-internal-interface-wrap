386 static constexpr hf_u32_t DutyCycleToRaw(float duty_cycle, hf_u8_t resolution_bits) noexcept {

388 duty_cycle = 0.0f;

400 static constexpr float RawToDutyCycle(hf_u32_t raw_value, hf_u8_t resolution_bits) noexcept {
401 hf_u32_t max_value = (1U << resolution_bits) - 1;
