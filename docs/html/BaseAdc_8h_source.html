320 hf_u8_t num_channels, hf_u32_t* readings,
321 float* voltages) noexcept {
322 if (!channel_ids || !readings || !voltages) {
323 return hf_adc_err_t::ADC_ERR_NULL_POINTER;
324 }
325
326 for (hf_u8_t i = 0; i < num_channels; ++i) {
327 hf_adc_err_t err = ReadChannel(channel_ids[i], readings[i], voltages[i]);
328 if (err != hf_adc_err_t::ADC_SUCCESS) {
329 return err;
330 }
331 }
332 return hf_adc_err_t::ADC_SUCCESS;
333 }
