268constexpr uint32_t HF_ESP32_ADC_DATA_BYTES_PER_CONV =
269 SOC_ADC_DIGI_DATA_BYTES_PER_CONV;
270constexpr uint32_t HF_ESP32_ADC_MIN_FRAME_SIZE = 64;
271constexpr uint32_t HF_ESP32_ADC_MAX_FRAME_SIZE = 1024;
272constexpr uint32_t HF_ESP32_ADC_DEFAULT_FRAME_SIZE = 256;

280inline constexpr uint32_t CalcFrameSize(uint32_t samples_per_frame,
281 uint32_t enabled_channels) noexcept {
282 return samples_per_frame * enabled_channels * HF_ESP32_ADC_DATA_BYTES_PER_CONV;
283}
284
292inline constexpr uint32_t CalcBufferPoolSize(uint32_t samples_per_frame, uint32_t enabled_channels,
293 uint32_t max_store_frames) noexcept {
294 return CalcFrameSize(samples_per_frame, enabled_channels) * max_store_frames;
295}
296
304inline constexpr bool IsValidContinuousConfig(uint32_t samples_per_frame, uint32_t enabled_channels,
305 uint32_t max_store_frames) noexcept {
306 if (enabled_channels == 0 || samples_per_frame == 0 || max_store_frames == 0) {
307 return false;
308 }
309
310 uint32_t frame_size = CalcFrameSize(samples_per_frame, enabled_channels);
311 uint32_t pool_size = CalcBufferPoolSize(samples_per_frame, enabled_channels, max_store_frames);
312
313 return (frame_size >= HF_ESP32_ADC_MIN_FRAME_SIZE) &&
314 (frame_size <= HF_ESP32_ADC_MAX_FRAME_SIZE) &&
315 (pool_size <= 32768) && // Reasonable max pool size (32KB)
316 ((frame_size % HF_ESP32_ADC_DATA_BYTES_PER_CONV) == 0);
317}
318
324inline constexpr bool IsValidFrameSize(uint32_t frame_size) noexcept {
325 return (frame_size >= HF_ESP32_ADC_MIN_FRAME_SIZE) &&
326 (frame_size <= HF_ESP32_ADC_MAX_FRAME_SIZE) &&
327 ((frame_size % HF_ESP32_ADC_DATA_BYTES_PER_CONV) == 0);
328}
329
335inline constexpr uint32_t GetFrameResultCount(uint32_t frame_size) noexcept {
336 return frame_size / HF_ESP32_ADC_DATA_BYTES_PER_CONV;
337}

350inline constexpr hf_channel_id_t GpioToAdcChannel(hf_pin_num_t gpio_pin) noexcept {
351 // Note: This is a simplified compile-time function.
352 // For accurate GPIO-to-channel conversion, use adc_continuous_io_to_channel()
353 // at runtime, which handles all ESP32 variants correctly.
354 return (gpio_pin >= 0 && gpio_pin <= 6) ? static_cast<hf_channel_id_t>(gpio_pin)
355 : HF_INVALID_CHANNEL;
356}

365inline constexpr hf_pin_num_t AdcChannelToGpio(hf_channel_id_t channel_id) noexcept {
366 // Note: This is a simplified compile-time function.
367 // For accurate channel-to-GPIO conversion, use adc_continuous_channel_to_io()
368 // at runtime, which handles all ESP32 variants correctly.
369 return (channel_id <= 6) ? static_cast<hf_pin_num_t>(channel_id) : HF_INVALID_PIN;
370}

377inline constexpr uint32_t GetMaxInputVoltage(hf_adc_atten_t atten) noexcept {
378 switch (atten) {
379 case hf_adc_atten_t::ATTEN_DB_0:
380 return 950; // ~0.95V
381 case hf_adc_atten_t::ATTEN_DB_2_5:
382 return 1320; // ~1.32V
383 case hf_adc_atten_t::ATTEN_DB_6:
384 return 1980; // ~1.98V
385 case hf_adc_atten_t::ATTEN_DB_12:
386 return 3300; // ~3.3V
387 default:
388 return 0;
389 }
390}

397inline constexpr uint32_t GetMaxRawValue(hf_adc_bitwidth_t bitwidth) noexcept {
398 switch (bitwidth) {
399 case hf_adc_bitwidth_t::WIDTH_9BIT:
400 return 511; // 2^9 - 1
401 case hf_adc_bitwidth_t::WIDTH_10BIT:
402 return 1023; // 2^10 - 1
403 case hf_adc_bitwidth_t::WIDTH_11BIT:
404 return 2047; // 2^11 - 1
405 case hf_adc_bitwidth_t::WIDTH_12BIT:
406 return 4095; // 2^12 - 1
407 default:
408 return 4095;
