631 virtual hf_u32_t SendMessageBatch(const hf_can_message_t* messages, hf_u32_t count,
632 hf_u32_t timeout_ms = 1000) noexcept {
633 if (!messages || count == 0) {
634 return 0;
635 }
636
637 hf_u32_t sent_count = 0;
638 for (hf_u32_t i = 0; i < count; ++i) {
639 if (SendMessage(messages[i], timeout_ms) == hf_can_err_t::CAN_SUCCESS) {
640 sent_count++;
641 }
642 }
643 return sent_count;
644 }
645
654 virtual hf_u32_t ReceiveMessageBatch(hf_can_message_t* messages, hf_u32_t max_count,
655 hf_u32_t timeout_ms = 100) noexcept {
656 if (!messages || max_count == 0) {
657 return 0;
658 }
659
660 hf_u32_t received_count = 0;
661 for (hf_u32_t i = 0; i < max_count; ++i) {
662 if (ReceiveMessage(messages[i], timeout_ms) == hf_can_err_t::CAN_SUCCESS) {
663 received_count++;
664 } else {
665 break; // No more messages available
666 }
667 }
668 return received_count;
669 }
