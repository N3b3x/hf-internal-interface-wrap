883inline hf_temp_err_t BaseTemperature::ReadTemperatureUnit(float* temperature,
884 hf_temp_unit_t unit) noexcept {
885 if (temperature == nullptr) {
886 return hf_temp_err_t::TEMP_ERR_NULL_POINTER;
887 }
888
889 float celsius;
890 hf_temp_err_t error = ReadTemperatureCelsius(&celsius);
891 if (error == hf_temp_err_t::TEMP_SUCCESS) {
892 switch (unit) {
893 case HF_TEMP_UNIT_CELSIUS:
894 *temperature = celsius;
895 break;
896 case HF_TEMP_UNIT_FAHRENHEIT:
897 *temperature = HF_TEMP_CELSIUS_TO_FAHRENHEIT(celsius);
898 break;
899 case HF_TEMP_UNIT_KELVIN:
900 *temperature = HF_TEMP_CELSIUS_TO_KELVIN(celsius);
901 break;
902 case HF_TEMP_UNIT_RANKINE:
903 *temperature = HF_TEMP_CELSIUS_TO_KELVIN(celsius) * 9.0f / 5.0f;
904 break;
905 default:
906 error = hf_temp_err_t::TEMP_ERR_INVALID_PARAMETER;
907 break;
908 }
909 }
910 return error;
