329 virtual hf_adc_err_t ReadMultipleChannels(const hf_channel_id_t* channel_ids,
330 hf_u8_t num_channels, hf_u32_t* readings,
331 float* voltages) noexcept {
332 if (!channel_ids || !readings || !voltages) {
333 return hf_adc_err_t::ADC_ERR_NULL_POINTER;
334 }
335
336 for (hf_u8_t i = 0; i < num_channels; ++i) {
337 hf_adc_err_t err = ReadChannel(channel_ids[i], readings[i], voltages[i]);
338 if (err != hf_adc_err_t::ADC_SUCCESS) {
339 return err;
340 }
341 }
342 return hf_adc_err_t::ADC_SUCCESS;
343 }
