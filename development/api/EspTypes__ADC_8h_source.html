269constexpr uint32_t HF_ESP32_ADC_DATA_BYTES_PER_CONV =
270 SOC_ADC_DIGI_DATA_BYTES_PER_CONV;
271constexpr uint32_t HF_ESP32_ADC_MIN_FRAME_SIZE = 64;
272constexpr uint32_t HF_ESP32_ADC_MAX_FRAME_SIZE = 1024;
273constexpr uint32_t HF_ESP32_ADC_DEFAULT_FRAME_SIZE = 256;

281inline constexpr uint32_t CalcFrameSize(uint32_t samples_per_frame,
282 uint32_t enabled_channels) noexcept {
283 return samples_per_frame * enabled_channels * HF_ESP32_ADC_DATA_BYTES_PER_CONV;
284}
285
293inline constexpr uint32_t CalcBufferPoolSize(uint32_t samples_per_frame, uint32_t enabled_channels,
294 uint32_t max_store_frames) noexcept {
295 return CalcFrameSize(samples_per_frame, enabled_channels) * max_store_frames;
296}
297
305inline constexpr bool IsValidContinuousConfig(uint32_t samples_per_frame, uint32_t enabled_channels,
306 uint32_t max_store_frames) noexcept {
307 if (enabled_channels == 0 || samples_per_frame == 0 || max_store_frames == 0) {
308 return false;
309 }
310
311 uint32_t frame_size = CalcFrameSize(samples_per_frame, enabled_channels);
312 uint32_t pool_size = CalcBufferPoolSize(samples_per_frame, enabled_channels, max_store_frames);
313
314 return (frame_size >= HF_ESP32_ADC_MIN_FRAME_SIZE) &&
315 (frame_size <= HF_ESP32_ADC_MAX_FRAME_SIZE) &&
316 (pool_size <= 32768) && // Reasonable max pool size (32KB)
317 ((frame_size % HF_ESP32_ADC_DATA_BYTES_PER_CONV) == 0);
318}
319
325inline constexpr bool IsValidFrameSize(uint32_t frame_size) noexcept {
326 return (frame_size >= HF_ESP32_ADC_MIN_FRAME_SIZE) &&
327 (frame_size <= HF_ESP32_ADC_MAX_FRAME_SIZE) &&
328 ((frame_size % HF_ESP32_ADC_DATA_BYTES_PER_CONV) == 0);
329}
330
336inline constexpr uint32_t GetFrameResultCount(uint32_t frame_size) noexcept {
337 return frame_size / HF_ESP32_ADC_DATA_BYTES_PER_CONV;
338}

351inline constexpr hf_channel_id_t GpioToAdcChannel(hf_pin_num_t gpio_pin) noexcept {
352 // Note: This is a simplified compile-time function.
353 // For accurate GPIO-to-channel conversion, use adc_continuous_io_to_channel()
354 // at runtime, which handles all ESP32 variants correctly.
355 return (gpio_pin >= 0 && gpio_pin <= 6) ? static_cast<hf_channel_id_t>(gpio_pin)
356 : HF_INVALID_CHANNEL;
357}

366inline constexpr hf_pin_num_t AdcChannelToGpio(hf_channel_id_t channel_id) noexcept {
367 // Note: This is a simplified compile-time function.
368 // For accurate channel-to-GPIO conversion, use adc_continuous_channel_to_io()
369 // at runtime, which handles all ESP32 variants correctly.
370 return (channel_id <= 6) ? static_cast<hf_pin_num_t>(channel_id) : HF_INVALID_PIN;
371}

378inline constexpr uint32_t GetMaxInputVoltage(hf_adc_atten_t atten) noexcept {
379 switch (atten) {
380 case hf_adc_atten_t::ATTEN_DB_0:
381 return 950; // ~0.95V
382 case hf_adc_atten_t::ATTEN_DB_2_5:
383 return 1320; // ~1.32V
384 case hf_adc_atten_t::ATTEN_DB_6:
385 return 1980; // ~1.98V
386 case hf_adc_atten_t::ATTEN_DB_12:
387 return 3300; // ~3.3V
388 default:
389 return 0;
390 }
391}

398inline constexpr uint32_t GetMaxRawValue(hf_adc_bitwidth_t bitwidth) noexcept {
399 switch (bitwidth) {
400 case hf_adc_bitwidth_t::WIDTH_9BIT:

406 case hf_adc_bitwidth_t::WIDTH_12BIT:
