638 virtual hf_u32_t SendMessageBatch(const hf_can_message_t* messages, hf_u32_t count,
639 hf_u32_t timeout_ms = 1000) noexcept {
640 if (!messages || count == 0) {
641 return 0;
642 }
643
644 hf_u32_t sent_count = 0;
645 for (hf_u32_t i = 0; i < count; ++i) {
646 if (SendMessage(messages[i], timeout_ms) == hf_can_err_t::CAN_SUCCESS) {
647 sent_count++;
648 }
649 }
650 return sent_count;
651 }

661 virtual hf_u32_t ReceiveMessageBatch(hf_can_message_t* messages, hf_u32_t max_count,
662 hf_u32_t timeout_ms = 100) noexcept {
663 if (!messages || max_count == 0) {
664 return 0;
665 }
666
667 hf_u32_t received_count = 0;
668 for (hf_u32_t i = 0; i < max_count; ++i) {
669 if (ReceiveMessage(messages[i], timeout_ms) == hf_can_err_t::CAN_SUCCESS) {
670 received_count++;
671 } else {
672 break; // No more messages available
673 }
674 }
675 return received_count;
676 }
