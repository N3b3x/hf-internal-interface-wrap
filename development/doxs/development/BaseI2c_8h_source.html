97constexpr std::string_view HfI2CErrToString(hf_i2c_err_t err) noexcept {
98 switch (err) {
99#define X(NAME, VALUE, DESC) \
100 case hf_i2c_err_t::NAME: \
101 return DESC;
102 HF_I2C_ERR_LIST(X)
103#undef X
104 default:
105 return HfI2CErrToString(hf_i2c_err_t::I2C_ERR_UNKNOWN);
106 }
