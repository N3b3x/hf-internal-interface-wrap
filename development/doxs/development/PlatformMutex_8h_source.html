199

206 }
207

211 }

214 Backend::unlockRecursive(&handle_);
215 }
216

219
220 // -- FreeRTOS-style convenience API ------------------------------------
221

223 return (timeout_ms > 0) ? try_lock_for(timeout_ms) : lock();
224 }
225
226 void Give() noexcept { unlock(); }
227

229
230 bool lock_shared() noexcept { return lock(); }
231 bool try_lock_shared() noexcept { return try_lock(); }
232 bool try_lock_shared_for(uint32_t timeout_ms) noexcept { return try_lock_for(timeout_ms); }
233 void unlock_shared() noexcept { unlock(); }

235private:

237};

251template <typename Backend = PlatformMutexActiveBackend>
252class PlatformSharedMutexImpl {
253public:
254 PlatformSharedMutexImpl() noexcept
255 : writer_mutex_{}, reader_mutex_{}, readers_(0), writer_active_(false) {
256 Backend::createMutex(&writer_mutex_);
257 Backend::createMutex(&reader_mutex_);
258 }
259
260 ~PlatformSharedMutexImpl() noexcept {
261 Backend::destroyMutex(&writer_mutex_);
262 Backend::destroyMutex(&reader_mutex_);
263 }
264
265 // Non-copyable
266 PlatformSharedMutexImpl(const PlatformSharedMutexImpl&) = delete;
267 PlatformSharedMutexImpl& operator=(const PlatformSharedMutexImpl&) = delete;
268
269 // Movable
270 PlatformSharedMutexImpl(PlatformSharedMutexImpl&& other) noexcept
271 : writer_mutex_(other.writer_mutex_), reader_mutex_(other.reader_mutex_),
272 readers_(other.readers_.load()), writer_active_(other.writer_active_.load()) {
273 other.writer_mutex_ = {};
274 other.reader_mutex_ = {};
275 other.readers_.store(0);
276 other.writer_active_.store(false);
277 }
278
279 PlatformSharedMutexImpl& operator=(PlatformSharedMutexImpl&& other) noexcept {
280 if (this != &other) {
281 Backend::destroyMutex(&writer_mutex_);
282 Backend::destroyMutex(&reader_mutex_);
283 writer_mutex_ = other.writer_mutex_;
284 reader_mutex_ = other.reader_mutex_;
285 readers_.store(other.readers_.load());
286 writer_active_.store(other.writer_active_.load());
287 other.writer_mutex_ = {};
288 other.reader_mutex_ = {};
289 other.readers_.store(0);
290 other.writer_active_.store(false);
291 }
292 return *this;
293 }
294
295 // -- Exclusive (writer) lock -------------------------------------------
296
297 bool lock() noexcept {
298 if (!Backend::lockMutex(&writer_mutex_, Backend::MAX_DELAY)) return false;
299 writer_active_.store(true);
300 while (readers_.load() > 0) { Backend::yield(); }
301 return true;
302 }
303
304 bool try_lock() noexcept {
305 if (!Backend::tryLockMutex(&writer_mutex_)) return false;
306 writer_active_.store(true);
307 if (readers_.load() > 0) {
308 writer_active_.store(false);
309 Backend::unlockMutex(&writer_mutex_);
310 return false;
311 }
312 return true;
313 }
314
315 bool try_lock_for(uint32_t timeout_ms) noexcept {
316 const uint32_t ticks = PlatformTime::MsToTicks(timeout_ms);
317 const uint32_t start = Backend::getTickCount();
318 if (!Backend::lockMutex(&writer_mutex_, ticks)) return false;
319 writer_active_.store(true);
320 const uint32_t elapsed = Backend::getTickCount() - start;
321 const uint32_t remaining = (elapsed < ticks) ? ticks - elapsed : 0;
322 const uint32_t deadline = Backend::getTickCount() + remaining;
323 while (readers_.load() > 0) {
324 if (Backend::getTickCount() >= deadline) {
325 writer_active_.store(false);
326 Backend::unlockMutex(&writer_mutex_);
327 return false;
328 }
329 Backend::yield();
330 }
331 return true;
332 }
333
334 void unlock() noexcept {
335 writer_active_.store(false);
336 Backend::unlockMutex(&writer_mutex_);
337 }
338
339 // -- Shared (reader) lock ----------------------------------------------
340
341 bool lock_shared() noexcept {
342 while (true) {
343 if (!Backend::lockMutex(&reader_mutex_, Backend::MAX_DELAY)) return false;
344 if (!writer_active_.load()) {
345 readers_++;
346 Backend::unlockMutex(&reader_mutex_);
347 return true;
348 }
349 Backend::unlockMutex(&reader_mutex_);
350 Backend::yield();
351 }
352 }
353
354 bool try_lock_shared() noexcept {
355 if (!Backend::tryLockMutex(&reader_mutex_)) return false;
356 if (!writer_active_.load()) {
357 readers_++;
358 Backend::unlockMutex(&reader_mutex_);
359 return true;
360 }
361 Backend::unlockMutex(&reader_mutex_);
362 return false;
363 }
364
365 bool try_lock_shared_for(uint32_t timeout_ms) noexcept {
366 const uint32_t ticks = PlatformTime::MsToTicks(timeout_ms);
367 const uint32_t start = Backend::getTickCount();
368 while (true) {
369 const uint32_t elapsed = Backend::getTickCount() - start;
370 if (elapsed >= ticks) return false;
371 const uint32_t remaining = ticks - elapsed;
372 if (!Backend::lockMutex(&reader_mutex_, remaining)) return false;
373 if (!writer_active_.load()) {
374 readers_++;
375 Backend::unlockMutex(&reader_mutex_);
376 return true;
377 }
378 Backend::unlockMutex(&reader_mutex_);
379 const uint32_t new_elapsed = Backend::getTickCount() - start;
380 if (new_elapsed >= ticks) return false;
381 Backend::yield();
382 }
383 }
384
385 void unlock_shared() noexcept {
386 if (Backend::lockMutex(&reader_mutex_, Backend::MAX_DELAY)) {
387 if (readers_.load() > 0) readers_--;
388 Backend::unlockMutex(&reader_mutex_);
389 }
390 }
391
392private:
393 typename Backend::MutexHandle writer_mutex_;
394 typename Backend::MutexHandle reader_mutex_;
395 std::atomic<int> readers_;
396 std::atomic<bool> writer_active_;
397};

418
