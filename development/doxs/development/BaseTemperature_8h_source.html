841inline hf_temp_err_t BaseTemperature::ReadTemperatureCelsius(float* temperature_celsius) noexcept {
842 if (temperature_celsius == nullptr) {
843 return hf_temp_err_t::TEMP_ERR_NULL_POINTER;
844 }
845
846 hf_temp_err_t validation = ValidateBasicOperation();
847 if (validation != hf_temp_err_t::TEMP_SUCCESS) {
848 return validation;
849 }
850
851 current_state_ = HF_TEMP_STATE_READING;
852 hf_temp_err_t result = ReadTemperatureCelsiusImpl(temperature_celsius);
853 current_state_ =
854 (result == hf_temp_err_t::TEMP_SUCCESS) ? HF_TEMP_STATE_INITIALIZED : HF_TEMP_STATE_ERROR;
855
856 return result;
857}

908inline hf_temp_err_t BaseTemperature::ReadTemperatureUnit(float* temperature,

912 }
913
914 float celsius;
915 hf_temp_err_t error = ReadTemperatureCelsius(&celsius);
916 if (error == hf_temp_err_t::TEMP_SUCCESS) {
917 switch (unit) {
918 case HF_TEMP_UNIT_CELSIUS:
919 *temperature = celsius;
920 break;
921 case HF_TEMP_UNIT_FAHRENHEIT:
922 *temperature = HF_TEMP_CELSIUS_TO_FAHRENHEIT(celsius);
923 break;
924 case HF_TEMP_UNIT_KELVIN:
925 *temperature = HF_TEMP_CELSIUS_TO_KELVIN(celsius);
926 break;
927 case HF_TEMP_UNIT_RANKINE:
928 *temperature = HF_TEMP_CELSIUS_TO_KELVIN(celsius) * 9.0f / 5.0f;
929 break;
930 default:
931 error = hf_temp_err_t::TEMP_ERR_INVALID_PARAMETER;
932 break;
933 }
934 }
