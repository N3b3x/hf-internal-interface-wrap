840 if (temperature_celsius == nullptr) {
841 return hf_temp_err_t::TEMP_ERR_NULL_POINTER;
842 }
843
844 hf_temp_err_t validation = ValidateBasicOperation();
845 if (validation != hf_temp_err_t::TEMP_SUCCESS) {
846 return validation;
847 }
848
849 current_state_ = HF_TEMP_STATE_READING;
850 hf_temp_err_t result = ReadTemperatureCelsiusImpl(temperature_celsius);
851 current_state_ =

853
854 return result;

924 break;
