936inline hf_gpio_err_t BaseGpio::Toggle() noexcept {
937 hf_gpio_err_t validation = ValidateBasicOperation();
938 if (validation != hf_gpio_err_t::GPIO_SUCCESS) {
939 return validation;
940 }
941
942 // Read current level, invert it, and set the new level
943 hf_gpio_level_t current_level;
944 hf_gpio_err_t result = GetPinLevelImpl(current_level);
945 if (result != hf_gpio_err_t::GPIO_SUCCESS) {
946 return result;
947 }
948
949 hf_gpio_level_t new_level = (current_level == hf_gpio_level_t::HF_GPIO_LEVEL_HIGH)
950 ? hf_gpio_level_t::HF_GPIO_LEVEL_LOW
951 : hf_gpio_level_t::HF_GPIO_LEVEL_HIGH;
952
953 result = SetPinLevelImpl(new_level);
954 if (result == hf_gpio_err_t::GPIO_SUCCESS) {
955 current_state_ = LevelToState(new_level);
956 }
957 return result;
958}

960inline hf_gpio_err_t BaseGpio::IsActive(bool& is_active) noexcept {

964 }

987inline hf_gpio_level_t BaseGpio::StateToLevel(hf_gpio_state_t state) const noexcept {
988 bool active_high = (active_state_ == hf_gpio_active_state_t::HF_GPIO_ACTIVE_HIGH);
989 bool should_be_high =
990 (state == hf_gpio_state_t::HF_GPIO_STATE_ACTIVE) ? active_high : !active_high;
991 return should_be_high ? hf_gpio_level_t::HF_GPIO_LEVEL_HIGH : hf_gpio_level_t::HF_GPIO_LEVEL_LOW;
992}
993
994inline hf_gpio_state_t BaseGpio::LevelToState(hf_gpio_level_t level) const noexcept {
995 bool active_high = (active_state_ == hf_gpio_active_state_t::HF_GPIO_ACTIVE_HIGH);
996 bool is_high = (level == hf_gpio_level_t::HF_GPIO_LEVEL_HIGH);
997 return (is_high == active_high) ? hf_gpio_state_t::HF_GPIO_STATE_ACTIVE
998 : hf_gpio_state_t::HF_GPIO_STATE_INACTIVE;
999}
