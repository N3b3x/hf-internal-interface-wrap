278constexpr uint32_t HF_ESP32_ADC_DATA_BYTES_PER_CONV =
279 SOC_ADC_DIGI_DATA_BYTES_PER_CONV;
280constexpr uint32_t HF_ESP32_ADC_MIN_FRAME_SIZE = 64;
281constexpr uint32_t HF_ESP32_ADC_MAX_FRAME_SIZE = 1024;
282constexpr uint32_t HF_ESP32_ADC_DEFAULT_FRAME_SIZE = 256;

290inline constexpr uint32_t CalcFrameSize(uint32_t samples_per_frame,
291 uint32_t enabled_channels) noexcept {
292 return samples_per_frame * enabled_channels * HF_ESP32_ADC_DATA_BYTES_PER_CONV;
293}
294
302inline constexpr uint32_t CalcBufferPoolSize(uint32_t samples_per_frame, uint32_t enabled_channels,
303 uint32_t max_store_frames) noexcept {
304 return CalcFrameSize(samples_per_frame, enabled_channels) * max_store_frames;
305}
306
314inline constexpr bool IsValidContinuousConfig(uint32_t samples_per_frame, uint32_t enabled_channels,
315 uint32_t max_store_frames) noexcept {
316 if (enabled_channels == 0 || samples_per_frame == 0 || max_store_frames == 0) {
317 return false;
318 }
319
320 uint32_t frame_size = CalcFrameSize(samples_per_frame, enabled_channels);
321 uint32_t pool_size = CalcBufferPoolSize(samples_per_frame, enabled_channels, max_store_frames);
322
323 return (frame_size >= HF_ESP32_ADC_MIN_FRAME_SIZE) &&
324 (frame_size <= HF_ESP32_ADC_MAX_FRAME_SIZE) &&
325 (pool_size <= 32768) && // Reasonable max pool size (32KB)
326 ((frame_size % HF_ESP32_ADC_DATA_BYTES_PER_CONV) == 0);
327}
328
334inline constexpr bool IsValidFrameSize(uint32_t frame_size) noexcept {
335 return (frame_size >= HF_ESP32_ADC_MIN_FRAME_SIZE) &&
336 (frame_size <= HF_ESP32_ADC_MAX_FRAME_SIZE) &&
337 ((frame_size % HF_ESP32_ADC_DATA_BYTES_PER_CONV) == 0);
338}
339
345inline constexpr uint32_t GetFrameResultCount(uint32_t frame_size) noexcept {
346 return frame_size / HF_ESP32_ADC_DATA_BYTES_PER_CONV;
347}

360inline constexpr hf_channel_id_t GpioToAdcChannel(hf_pin_num_t gpio_pin) noexcept {
361 // Note: This is a simplified compile-time function.
362 // For accurate GPIO-to-channel conversion, use adc_continuous_io_to_channel()
363 // at runtime, which handles all ESP32 variants correctly.
364 return (gpio_pin >= 0 && gpio_pin <= 6) ? static_cast<hf_channel_id_t>(gpio_pin)
365 : HF_INVALID_CHANNEL;
366}

375inline constexpr hf_pin_num_t AdcChannelToGpio(hf_channel_id_t channel_id) noexcept {
376 // Note: This is a simplified compile-time function.
377 // For accurate channel-to-GPIO conversion, use adc_continuous_channel_to_io()
378 // at runtime, which handles all ESP32 variants correctly.
379 return (channel_id <= 6) ? static_cast<hf_pin_num_t>(channel_id) : HF_INVALID_PIN;
380}

387inline constexpr uint32_t GetMaxInputVoltage(hf_adc_atten_t atten) noexcept {
388 switch (atten) {
389 case hf_adc_atten_t::ATTEN_DB_0:
390 return 950; // ~0.95V
391 case hf_adc_atten_t::ATTEN_DB_2_5:
392 return 1320; // ~1.32V
393 case hf_adc_atten_t::ATTEN_DB_6:
394 return 1980; // ~1.98V
395 case hf_adc_atten_t::ATTEN_DB_12:
396 return 3300; // ~3.3V
397 default:
398 return 0;
399 }
400}

407inline constexpr uint32_t GetMaxRawValue(hf_adc_bitwidth_t bitwidth) noexcept {
408 switch (bitwidth) {

411 case hf_adc_bitwidth_t::WIDTH_10BIT:

415 case hf_adc_bitwidth_t::WIDTH_12BIT:
416 return 4095; // 2^12 - 1
