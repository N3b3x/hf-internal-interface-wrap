205inline constexpr size_t GetMaxValueSizeForType(hf_nvs_type_t type) noexcept {
206 switch (type) {
207 case hf_nvs_type_t::HF_NVS_TYPE_U8:
208 case hf_nvs_type_t::HF_NVS_TYPE_I8:
209 return 1;
210 case hf_nvs_type_t::HF_NVS_TYPE_U16:
211 case hf_nvs_type_t::HF_NVS_TYPE_I16:
212 return 2;
213 case hf_nvs_type_t::HF_NVS_TYPE_U32:
214 case hf_nvs_type_t::HF_NVS_TYPE_I32:
215 return 4;
216 case hf_nvs_type_t::HF_NVS_TYPE_U64:
217 case hf_nvs_type_t::HF_NVS_TYPE_I64:
218 return 8;
219 case hf_nvs_type_t::HF_NVS_TYPE_STR:
220 case hf_nvs_type_t::HF_NVS_TYPE_BLOB:
221 return HF_NVS_MAX_VALUE_SIZE;
222 default:
223 return 0;
224 }
