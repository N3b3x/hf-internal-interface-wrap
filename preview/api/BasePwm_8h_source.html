400 static constexpr hf_u32_t DutyCycleToRaw(float duty_cycle, hf_u8_t resolution_bits) noexcept {
401 if (duty_cycle < 0.0f)
402 duty_cycle = 0.0f;
403 if (duty_cycle > 1.0f)
404 duty_cycle = 1.0f;
405 return static_cast<hf_u32_t>(duty_cycle * ((1U << resolution_bits) - 1));
406 }

414 static constexpr float RawToDutyCycle(hf_u32_t raw_value, hf_u8_t resolution_bits) noexcept {
415 hf_u32_t max_value = (1U << resolution_bits) - 1;
416 if (raw_value > max_value)
417 raw_value = max_value;
418 return static_cast<float>(raw_value) / static_cast<float>(max_value);
419 }

448 static constexpr bool IsValidRawDuty(hf_u32_t raw_value, hf_u8_t resolution_bits) noexcept {
449 if (resolution_bits == 0 || resolution_bits > 16)
450 return false;

461 static constexpr float CalculateFrequencyAccuracy(hf_u32_t target_freq,
462 hf_u32_t actual_freq) noexcept {
463 if (target_freq == 0)
464 return 0.0f;
465 float diff = static_cast<float>(target_freq > actual_freq ? target_freq - actual_freq
466 : actual_freq - target_freq);
467 return 1.0f - (diff / static_cast<float>(target_freq));
468 }
