412inline bool isValidWifiPassword(const std::string& password, hf_wifi_security_t security) {
413 if (security == hf_wifi_security_t::HF_WIFI_SECURITY_OPEN) {
414 return password.empty();
415 }
416
417 if (security == hf_wifi_security_t::HF_WIFI_SECURITY_WEP) {
418 // WEP key length constants
419 static constexpr size_t WEP_KEY_64_BIT_ASCII = 5; // 64-bit WEP ASCII key
420 static constexpr size_t WEP_KEY_128_BIT_ASCII = 13; // 128-bit WEP ASCII key
421 static constexpr size_t WEP_KEY_128_BIT_HEX = 16; // 128-bit WEP hexadecimal key
422 static constexpr size_t WEP_KEY_152_BIT_ASCII = 29; // 152-bit WEP ASCII key
423
424 size_t len = password.length();
425 return (len == WEP_KEY_64_BIT_ASCII || len == WEP_KEY_128_BIT_ASCII ||
426 len == WEP_KEY_128_BIT_HEX || len == WEP_KEY_152_BIT_ASCII);
427 }
428
429 // WPA/WPA2/WPA3 passwords must be 8-63 characters
430 return (password.length() >= 8 && password.length() <= 63);
431}

453inline void hfStationConfigToEspConfig(const hf_wifi_station_config_t& hf_config,
454 wifi_sta_config_t& esp_config) {
455 std::memset(&esp_config, 0, sizeof(esp_config));
456
457 // Copy SSID
458 std::strncpy(reinterpret_cast<char*>(esp_config.ssid), hf_config.ssid.c_str(),
459 sizeof(esp_config.ssid) - 1);
460
461 // Copy password
462 std::strncpy(reinterpret_cast<char*>(esp_config.password), hf_config.password.c_str(),
463 sizeof(esp_config.password) - 1);
464
465 // Copy BSSID if set
466 if (hf_config.bssid_set) {
467 std::memcpy(esp_config.bssid, hf_config.bssid, 6);
468 esp_config.bssid_set = true;
469 }
470
471 // Set channel
472 esp_config.channel = hf_config.channel;
473
474 // Set scan method
475 esp_config.scan_method = static_cast<wifi_scan_method_t>(hf_config.scan_method);
476
477 // Set sort method
478 esp_config.sort_method = static_cast<wifi_sort_method_t>(hf_config.sort_method);
479
480 // Set thresholds
481 esp_config.threshold.rssi = hf_config.threshold_rssi;
482 esp_config.threshold.authmode = hfWifiSecurityToEspAuthMode(hf_config.threshold_authmode);
483}
